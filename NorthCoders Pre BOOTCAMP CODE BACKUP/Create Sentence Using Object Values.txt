function createSentence(person) {
 
 return ('Hello my name is ' +  person.name + ", " +  "I am " + person.age + " years old" + " and I am a " + person.job )
 }


createSentence({ name: 'Mitch', age: 29, job: 'tutor' });

Using Template Literals
function createSentence(person) {
 
 return (`Hello my name is ${person.name}, and I am ${person.age} years old and I am a ${person.job}`)
 }
createSentence({ name: 'Mitch', age: 29, job: 'tutor' });
