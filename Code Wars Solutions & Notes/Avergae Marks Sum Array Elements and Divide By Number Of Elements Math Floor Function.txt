function getAverage(marks){

  //Setup a counter variable to store the value of all the elements stored in the marks array
  let counter = 0;
  // variable to store the number of elements in the array
  let elements = 0;
  //variable to store the result
  let result =0;
  //for loop to ierate through the array
  for (i=0; i<marks.length; i++){
    //count equals the previous value of itself plus the value stored in the current index of the array, the for loop
    //iterates using i so we pass the i value to the marks array as a call for each of its elements
    counter = counter + marks[i]  
    //Each time the loop executes we add the value of one to the elements variable this becomes a counter for the number
    //of elements in the array
    elements = elements +1
     //console.log(counter)
    //Then we perform the final math, using floor to round down the number to the closest full interger and deviding counter
    //by elements and storing the value in the result variable
    result = Math.floor(counter/elements)
  }
  //return the value of result which is the mean value of the grades
  return  (result)
 
}
