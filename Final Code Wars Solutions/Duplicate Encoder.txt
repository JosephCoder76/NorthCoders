function duplicateEncode(word){

//lowercase all the characters in the string so they all are the same from a comparison point of view
  
   word=word.toLowerCase();

// will be {"char": "number of occurance of each seperate character in the string"} Each
 //character will be stored as an object array
 
  characterOccurance = {} 

  /*For the word parameter/string we use the split method and the "" in the brackets instruct the split method to
  split on and individual character basis, the ForEach which calls a function once for each element in an array, in this case
  we will be calling the char function which will ensure that each of the characters returned from the array will be represented
  in a string format, this is useful as it prevents unintennded type mismatch and or conversion the => operator is instructing
  that the value of the current char in the element of the array must equal to or greater than "" so it basically has a single
  character value, this forEach loop will repeat until we have iterated throughout the entire word array or string
  */

  //Split the string by characters and for each character  
  word.split('').forEach(char => { 

   // increase the occurance count, else we dont find another occurance of the character in the string so we set the characterOccurance value to 1
    if(characterOccurance[char]){ 
       characterOccurance[char] += 1;

      }else{
       characterOccurance[char] = 1;
  }
  })
  
  // Here we decalre the local variable of output to equal the
  //result of using the split method on the word array for each character that does not equal 1 occurance we output ( else
  //we output ) this logical is predefined by the ?, the .map function creates a new array using the parent function which
  //will be the result of the word.split function, and then the .join function creates a new string by concatenating all
  //the elements of the array, so for each character that => equal to or greater than 1 characterOccurance(current character)
  //we add ( to the value of output, else false and we add )
  let output = word.split('').map(
                    char => characterOccurance[char] == 1 ? '(' :')' 
                    ).join('');
 
  return output;

}
